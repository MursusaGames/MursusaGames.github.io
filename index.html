using UnityEngine;
using UnityEngine.UI;
public class SettingMenuScript : MonoBehaviour
{
[SerializeField] GameObject onMusic_Btn;
[SerializeField] GameObject onSound_Btn;
private void Start()
{
if (PlayerPrefs.HasKey("OnMusic"))
{
OnMusicBtnChangeColor();
OnSoundBtnChangeColor();
}
onMusic += OnMusicBtnChangeColor;
onSound += OnSoundBtnChangeColor;
}
#region Events
public delegate void SettingsMenu();
public event SettingsMenu onMusic;
public event SettingsMenu onSound;
#endregion
#region Buttons
//кнопка OnMusic
public void OnMusic()
{
onMusic?.Invoke();
}
//кнопка OnSound
public void OnSound()
{
onSound?.Invoke();
}
#endregion
public void OnMusicBtnChangeColor()
{
if(PlayerPrefs.GetString("OnMusic") == "Yes")
{
onMusic_Btn.GetComponent().color = Color.green;
}
else
{
onMusic_Btn.GetComponent().color = Color.red;
}
}
public void OnSoundBtnChangeColor()
{
if (PlayerPrefs.GetString("OnSound") == "Yes")
{
onSound_Btn.GetComponent().color = Color.green;
}
else
{
onSound_Btn.GetComponent().color = Color.red;
}
}
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
public class GameManager : Singleton<|GameManager|>
{
[Header("Set in Inspector")]
public GameObject shipPrefab;
public Transform shipStartPoint;
public GameObject currentShip { get; private set; }
public GameObject spaceStationPrefab;
public Transform stationStartPoint;
public GameObject currentStation { get; private set; }
[Header("Set dinamycally")]
public SmoothFollow cameraFolow;
public GameObject gameOverUI;
public GameObject mainMenuUI;
public GameObject pausedUI;
public GameObject inGameUI;
public bool isGamePlayed { get; private set; }
public AsteroidSpawner asteroidSpawner;
public bool paused;
// Start is called before the first frame update void Start()
{
ShowMainMenu();
}
void ShowUI(GameObject newUI) //деактивирует все окна кроме полученного
{
GameObject[] allUI = { mainMenuUI, inGameUI, pausedUI, gameOverUI };
foreach(GameObject uiToHide in allUI)
{
uiToHide.SetActive(false);
}
newUI.SetActive(true);
}
public void ShowMainMenu() //заставка
{
ShowUI(mainMenuUI);
isGamePlayed = false;
asteroidSpawner.isSpawnAsteroids = false;
}
public void StartGame() //при нажатии кнопки старт игры в заставке стартует игра
{
ShowUI(inGameUI);
isGamePlayed = true;
if(currentShip!= null)
{
Destroy(currentShip);
}
if(currentStation != null)
{
Destroy(currentStation);
}
currentShip = Instantiate(shipPrefab);
currentShip.transform.position = shipStartPoint.position;
currentShip.transform.rotation = shipStartPoint.rotation;
currentStation = Instantiate(spaceStationPrefab);
currentStation.transform.position = stationStartPoint.position;
currentStation.transform.rotation = stationStartPoint.rotation;
cameraFolow.target = currentShip.transform;
asteroidSpawner.isSpawnAsteroids = true;
asteroidSpawner.target = currentStation.transform;

}
public void GameOver()//конец игры вызывается при разрушении станции
{
ShowUI(gameOverUI);
isGamePlayed = false;
if(currentShip != null)
{
Destroy(currentShip);
}
if(currentStation != null)
{
Destroy(currentStation);
}
asteroidSpawner.isSpawnAsteroids = false;
asteroidSpawner.DestroyAllAsteroids();
}
public void SetPaused(bool paused) //пауза
{
inGameUI.SetActive(!paused);
pausedUI.SetActive(paused);
if (paused)
{
Time.timeScale = 0.0f;
}
else
{
Time.timeScale = 1.0f;
}
}
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.EventSystems;

public class Player : MonoBehaviour, IDragHandler, IBeginDragHandler
{
public float speed = 5.0f;
public float step = 1.3f;
public float jumpHeight = 2.5f;
public bool isLeft = true;
public GameObject player;
void TurnUp()
{
Vector3 playerPos = player.transform.position;
if (playerPos.y < 0.5f)
{
playerPos.y = jumpHeight;
player.transform.position = playerPos;
}
}
public void OnBeginDrag(PointerEventData eventData)
{
Vector2 delta = eventData.delta; //фиксируем дельту тачеи
if (Mathf.Abs(delta.x) > Mathf.Abs(delta.y)) //тач по горизонтали
{
if (delta.x > 0) //свеип вправо
{
TurnRight();
}
else //свеип влево
{
TurnLeft();
}
}
else //тач по вертикали
{
if (delta.y > 0) //свеип вверх
{
TurnUp();
}
else
{
Debug.Log("Down");
}
}
}
public void OnDrag(PointerEventData eventData)
{
}
void TurnLeft()
{
if (!isLeft)
{
Vector3 playerPos = player.transform.position;
playerPos.x = -step;
player.transform.position = playerPos;
isLeft = true;
}
}
void TurnRight()
{
if (isLeft)
{
Vector3 playerPos = player.transform.position;
playerPos.x = step;
player.transform.position = playerPos;
isLeft = false;
}
}
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
public class PlayerController : MonoBehaviour
{
public float speed;
public float jumpForce;
private float moveInput;
private bool facingRight = true;
public bool isGrounded;
public Transform feetPos;
public float chekRadius;
public LayerMask whatIsGround;
public VectorValue pos;
Rigidbody2D rb;
Animator anim;
// Start is called before the first frame update
void Start()
{
transform.position = pos.initialValue;
rb = GetComponent();
anim = GetComponent();
}
// Update is called once per frame
void FixedUpdate()
{
moveInput = Input.GetAxis("Horizontal");
rb.velocity = new Vector2(moveInput * speed, rb.velocity.y);
if(facingRight == false && moveInput > 0) //поворот в другую сторону
{
Flip();
}
else if(facingRight == true && moveInput < 0)
{
Flip();
}
//анимация
if(moveInput == 0)
{
anim.SetBool("isRunning", false);
}
else
{
anim.SetBool("isRunning", true);
}
}
private void Update()
{
//проверяем землю
isGrounded = Physics2D.OverlapCircle(feetPos.position, chekRadius, whatIsGround);
if(isGrounded == true && Input.GetKeyDown(KeyCode.UpArrow)) //прыгаем {
rb.velocity = Vector2.up * jumpForce;
anim.SetTrigger("takeOff");
}
else if(isGrounded)
{
anim.SetBool("isJump", false);
}
else if(!isGrounded)
{
anim.SetBool("isJump", true);
}
}
void Flip()
{
facingRight = !facingRight;
Vector3 Scaler = transform.localScale;
Scaler.x *= -1;
transform.localScale = Scaler;
if(moveInput < 0)
{
transform.eulerAngles = new Vector3(0, 180, 0);
}
else if(moveInput > 0)
{
transform.eulerAngles = new Vector3(0, 0, 0);
}
}
}

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
public class Enemy : MonoBehaviour
{
[Header("Fire Block")]
public GameObject bulletPrefab;
public GameObject explosionPrefab;
public bool isFire = true;
public float timeBetwenFire = 9f;
public float stopFire;
public Transform thisTransform;
GameObject target;
public float rotationSpeed = 100f;
private void Awake()
{
thisTransform = GetComponent();
target = GameObject.FindGameObjectWithTag("Player");
}
IEnumerator TrackRotation(GameObject target)
{
while (true)
{
if(thisTransform != null && target != null)
{
Vector3 relativePos = target.transform.position - thisTransform.position;
Quaternion newRotation = Quaternion.LookRotation(relativePos);
thisTransform.rotation = Quaternion.RotateTowards(thisTransform.rotation, newRotation, rotationSpeed * Time.deltaTime);
yield return null;
}
}
}
void Update()
{
StartCoroutine(TrackRotation(target));
#region Время между выстрелами
if (isFire == false) //время между выстрелами
{
stopFire -= Time.deltaTime;
if (stopFire <= 0)
{
isFire = true;
stopFire = timeBetwenFire;
}
}
#endregion
#region Выстрел
if (isFire)
{
Instantiate(bulletPrefab, startBulletPosition.transform.position, startBulletPosition.transform.rotation);
Instantiate(explosionPrefab, startBulletPosition.transform.position, startBulletPosition.transform.rotation);
isFire = false;
}
#endregion
}
}
