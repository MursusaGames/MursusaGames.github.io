ПРИМЕРЫ КОДА

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class HiScore : MonoBehaviour
{
static public int score = 100;

// Если в PlayerPrefs есть HighScore считываем , иначе записыаем
private void Awake()
{
if (PlayerPrefs.HasKey("HighScore"))
{
score = PlayerPrefs.GetInt("HighScore");
}
PlayerPrefs.SetInt("HighScore", score);
}
// Update is called once per frame
void Update()
{
//Выводим на экран
Text gt = gameObject.GetComponent();
gt.text = "High Score: " + score;
//score увеличивается в другом скрипте здесь пишем её значение в PlayerPrefs если она больше чем HighScore
if(score > PlayerPrefs.GetInt("HighScore"))
{
PlayerPrefs.SetInt("HighScore", score);
}
}
}
